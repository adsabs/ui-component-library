## Install

```bash
npm install --save shared-react-bbb
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'shared-react-bbb'
import 'shared-react-bbb/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
